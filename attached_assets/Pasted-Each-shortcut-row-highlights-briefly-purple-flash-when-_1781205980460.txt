LEVEL COMPLETION — tracked in a dedicated table:

```sql
create table public.level_completion (
  id            uuid default uuid_generate_v4() primary key,
  user_id       uuid references public.profiles(id) on delete cascade,
  module        text not null,
  level_name    text not null,
  completed_at  timestamptz not null default now(),
  stars         integer not null default 1 check (stars between 1 and 3),
  total_xp      integer not null default 0,
  attempts      integer not null default 0,
  best_score    integer not null default 0,
  unique(user_id, module, level_name)
);
```

Stars calculation:
  3 stars: completed all challenges with 0 hints and 0 retries
  2 stars: completed with 1–3 hints or 1 retry
  1 star:  completed with 4+ hints or 2+ retries

Level is marked complete when:
  Shortcut module: 100% of challenges in that level completed
  Formatting module: all challenges in that level completed
  (not 70% — completion means ALL done, 70% unlocks next level)

WHAT TO SAVE ON EACH EVENT:

1. Challenge started:
   INSERT activity_log: action='started'

2. Challenge completed:
   INSERT activity_log: action='completed', xp_earned, score, time_taken_sec
   UPSERT shortcut_progress or formatting_progress: status='completed', bestXP
   UPSERT last_challenge table
   CHECK if level now complete → if yes:
     INSERT level_completion (or UPDATE if revisiting)
     Award level completion XP bonus:
       Rookie: +100 XP
       Intermediate: +200 XP
       Advanced: +300 XP
       Expert: +500 XP
     Trigger badge check

3. Hint used:
   UPDATE shortcut_progress: hints_used += 1

4. Challenge failed (3 wrong, timeout):
   INSERT activity_log: action='failed'

5. Retry:
   INSERT activity_log: action='retried'

6. Skip:
   INSERT activity_log: action='skipped'
   UPDATE status to 'review' in progress table

7. Study tab marked:
   INSERT activity_log: action='studied'
   UPDATE status to 'studied' in progress table

-- ─────────────────────────────────────────
create or replace function public.update_streak(p_user_id uuid)
returns void language plpgsql security definer as $$
declare
  v_last_active date;
  v_today date := current_date;
begin
  select last_active into v_last_active
  from public.profiles where id = p_user_id;

  if v_last_active = v_today then
    return; -- already logged in today
  elsif v_last_active = v_today - interval '1 day' then
    -- consecutive day
    update public.profiles
    set streak_count = streak_count + 1,
        last_active  = v_today,
        updated_at   = now()
    where id = p_user_id;
  else
    -- streak broken
    update public.profiles
    set streak_count = 1,
        last_active  = v_today,
        updated_at   = now()
    where id = p_user_id;
  end if;
end;
$$;

select
  p.id,

  p.avatar_url,

  p.streak_count,

// Log any challenge action
export async function logActivity(params: {
  userId: string
  module: 'shortcuts' | 'formatting'
  levelName: string
  challengeId: string
  challengeTitle: string
  action: 'started'|'completed'|'failed'|'retried'|'skipped'|'studied'
  xpEarned?: number
  score?: number
  hintsUsed?: number
  timeTakenSec?: number
  comboAtTime?: number
}) {
  await supabase.from('activity_log').insert({
    user_id:         params.userId,
    module:          params.module,
    level_name:      params.levelName,
    challenge_id:    params.challengeId,
    challenge_title: params.challengeTitle,
    action:          params.action,
    xp_earned:       params.xpEarned ?? 0,
    score:           params.score,
    hints_used:      params.hintsUsed ?? 0,
    time_taken_sec:  params.timeTakenSec,
    combo_at_time:   params.comboAtTime ?? 0,
  })
}

// Award XP and log transaction
export async function awardXP(params: {
  userId: string
  amount: number
  source: string
  sourceId?: string
  module: string
}) {
  await supabase.from('xp_transactions').insert({
    user_id:   params.userId,
    amount:    params.amount,
    source:    params.source,
    source_id: params.sourceId,
    module:    params.module,
  })
  // trigger on_xp_earned updates profiles.total_xp automatically
}

// Mark challenge complete, check level completion
export async function completeChallenge(params: {
  userId: string
  module: 'shortcuts' | 'formatting'
  levelName: string
  challengeId: string
  challengeTitle: string
  xpEarned: number
  score?: number
  hintsUsed: number
  timeTakenSec: number
  combo: number
  bestXP?: number
}) {
  const table = params.module === 'shortcuts'
    ? 'shortcut_progress' : 'formatting_progress'

  await supabase.from(table).upsert({
    user_id:      params.userId,
    [`${params.module === 'shortcuts' ? 'shortcut' : 'challenge'}_id`]:
                  params.challengeId,
    level_name:   params.levelName,
    status:       'completed',
    best_xp:      params.bestXP ?? params.xpEarned,
    hints_used:   params.hintsUsed,
    completed_at: new Date().toISOString(),
    updated_at:   new Date().toISOString(),
  }, { onConflict: 'user_id,' +
    (params.module === 'shortcuts' ? 'shortcut_id' : 'challenge_id') })

  await logActivity({
    userId:         params.userId,
    module:         params.module,
    levelName:      params.levelName,
    challengeId:    params.challengeId,
    challengeTitle: params.challengeTitle,
    action:         'completed',
    xpEarned:       params.xpEarned,
    score:          params.score,
    hintsUsed:      params.hintsUsed,
    timeTakenSec:   params.timeTakenSec,
    comboAtTime:    params.combo,
  })

  await awardXP({
    userId:   params.userId,
    amount:   params.xpEarned,
    source:   params.challengeTitle,
    sourceId: params.challengeId,
    module:   params.module,
  })

  await supabase.from('last_challenge').upsert({
    user_id:         params.userId,
    module:          params.module,
    challenge_id:    params.challengeId,
    challenge_title: params.challengeTitle,
    level_name:      params.levelName,
    xp_reward:       params.xpEarned,
    last_seen:       new Date().toISOString(),
  }, { onConflict: 'user_id' })
}

// Check and record level completion
export async function checkLevelCompletion(params: {
  userId: string
  module: 'shortcuts' | 'formatting'
  levelName: string
  totalInLevel: number
  hintsUsed: number
  retries: number
}) {
  const table = params.module === 'shortcuts'
    ? 'shortcut_progress' : 'formatting_progress'

  const { count } = await supabase
    .from(table)
    .select('*', { count: 'exact', head: true })
    .eq('user_id', params.userId)
    .eq('level_name', params.levelName)
    .eq('status', 'completed')

  if ((count ?? 0) < params.totalInLevel) return false

  const stars = params.hintsUsed === 0 && params.retries === 0 ? 3
    : params.hintsUsed <= 3 && params.retries <= 1 ? 2 : 1

  const bonusXP = {
    rookie: 100, intermediate: 200, advanced: 300, expert: 500
  }[params.levelName.toLowerCase()] ?? 100

  await supabase.from('level_completion').upsert({
    user_id:    params.userId,
    module:     params.module,
    level_name: params.levelName,
    stars,
    total_xp:   bonusXP,
  }, { onConflict: 'user_id,module,level_name' })

  await awardXP({
    userId:   params.userId,
    amount:   bonusXP,
    source:   `${params.levelName} level complete`,
    sourceId: `${params.module}_${params.levelName}`,
    module:   'level_complete',
  })

  return true
}

// Fetch user's full progress state
export async function fetchUserProgress(userId: string) {
  const [
    { data: profile },
    { data: shortcuts },
    { data: formatting },
    { data: levelsDone },
    { data: badges },
    { data: lastChallenge },
    { data: activity },
  ] = await Promise.all([
    supabase.from('profiles').select('*').eq('id', userId).single(),
    supabase.from('shortcut_progress').select('*').eq('user_id', userId),
    supabase.from('formatting_progress').select('*').eq('user_id', userId),
    supabase.from('level_completion').select('*').eq('user_id', userId),
    supabase.from('user_badges')
      .select('*, badges(*)').eq('user_id', userId),
    supabase.from('last_challenge').select('*').eq('user_id', userId).single(),
    supabase.from('activity_log')
      .select('*').eq('user_id', userId)
      .order('created_at', { ascending: false }).limit(20),
  ])

  return { profile, shortcuts, formatting, levelsDone,
           badges, lastChallenge, activity }
}

// Fetch leaderboard top 20
export async function fetchLeaderboard() {
  const { data } = await supabase
    .from('leaderboard')
    .select('*')
    .limit(20)
  return data
}
