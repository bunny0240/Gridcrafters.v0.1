Ctrl+! requires Shift+1. Browser intercepts Ctrl+Shift combos and
switches tabs. Fix in useShortcutDetection and useMultiShortcutDetection:

In the keydown handler, BEFORE any other logic:

  const dangerousCombos = [
    e.ctrlKey && e.shiftKey && ['Tab','PageDown','PageUp'].includes(e.key),
    e.ctrlKey && e.key === 'Tab',
    e.altKey && e.key === 'F4',
    e.ctrlKey && ['w','W','t','T','n','N'].includes(e.key),
  ]
  if (dangerousCombos.some(Boolean)) {
    e.preventDefault()
    e.stopImmediatePropagation()
    // for tab-switch specifically, still process as shortcut below
  }

For Ctrl+! detection, user physically presses Ctrl+Shift+1.
Browser fires e.key='!' when Shift+1 is held.
buildChord must produce ['Ctrl','!'] not ['Ctrl','Shift','1']:

  function buildChord(e: KeyboardEvent): string[] {
    const key = normalizeKey(e)
    // symbols produced by Shift+number already encode the shift
    const isShiftSymbol = /^[!@#$%^&*()_+~`<>?:"{}|]$/.test(key)
    const chord: string[] = []
    if (e.ctrlKey) chord.push('Ctrl')
    if (e.shiftKey && !isShiftSymbol) chord.push('Shift')
    if (e.altKey) chord.push('Alt')
    if (!['Ctrl','Shift','Alt'].includes(key)) chord.push(key)
    return chord
  }

Shortcut data entry for Ctrl+!:
  keys: ['Ctrl','!']   // matches what buildChord returns
