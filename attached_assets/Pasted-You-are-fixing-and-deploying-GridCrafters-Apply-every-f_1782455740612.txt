Create src/store/userStore.ts using React Context + useReducer:

  interface UserStore {
    profile: Profile | null
    loading: boolean
    error: string | null
  }

  const UserContext = createContext<{
    state: UserStore
    dispatch: Dispatch<Action>
  }>(null!)

  export function UserProvider({ children }) {
    const [state, dispatch] = useReducer(reducer, initialState)
    const { user } = useAuth()

    useEffect(() => {
      if (!user) return

      // Initial fetch
      fetchProfile(user.id).then(profile =>
        dispatch({ type: 'SET_PROFILE', payload: profile })
      )

      // Realtime subscription — fires on every profiles row update
      const channel = supabase
        .channel(`profile-${user.id}`)
        .on('postgres_changes', {
          event: 'UPDATE',
          schema: 'public',
          table: 'profiles',
          filter: `id=eq.${user.id}`
        }, (payload) => {
          dispatch({ type: 'SET_PROFILE', payload: payload.new as Profile })
        })
        .subscribe()

      return () => { supabase.removeChannel(channel) }
    }, [user])

    return (
      <UserContext.Provider value={{ state, dispatch }}>
        {children}
      </UserContext.Provider>
    )
  }

  export const useUserStore = () => useContext(UserContext)

Replace update_streak function entirely:

  create or replace function public.update_streak(p_user_id uuid)
  returns jsonb language plpgsql security definer as $$
  declare
    v_last    date;
    v_streak  integer;
    v_max     integer;
    v_today   date := current_date;
    v_result  jsonb;
  begin
    select last_active, streak_count, max_streak
    into v_last, v_streak, v_max
    from public.profiles where id = p_user_id;

    if v_last = v_today then
      -- Already visited today, no change
      return jsonb_build_object(
        'streak', v_streak, 'max_streak', v_max, 'updated', false
      );
    end if;

    if v_last = v_today - interval '1 day' then
      -- Consecutive day — increment
      v_streak := v_streak + 1;
    else
      -- Streak broken — reset to 1
      v_streak := 1;
    end if;

    -- Update max if current exceeds it
    if v_streak > v_max then
      v_max := v_streak;
    end if;

    update public.profiles set
      streak_count = v_streak,
      max_streak   = v_max,
      last_active  = v_today,
      updated_at   = now()
    where id = p_user_id;

    return jsonb_build_object(
      'streak', v_streak, 'max_streak', v_max, 'updated', true
    );
  end;
  $$;

  // When Shift is held, number keys produce symbols
  // e.key gives the symbol directly
  const shiftSymbols: Record<string, string> = {
    '!': '!', '@': '@', '#': '#', '$': '$',
    '%': '%', '^': '^', '&': '&', '*': '*',
    '(': '(', ')': ')', '_': '_', '+': '+',
    '~': '~', '<': '<', '>': '>', '?': '?',
    ':': ':', '"': '"', '{': '{', '}': '}',
    '|': '|',
  }
  if (shiftSymbols[e.key]) return e.key

Fix buildChordArray — when e.shiftKey is true AND the key is a
symbol (not a letter), do NOT add 'Shift' to the chord array
separately, because the symbol already encodes the shift:

  function buildChord(e: KeyboardEvent): string[] {
    const key = normalizeKey(e)
    const isSymbolFromShift = /^[!@#$%^&*()_+~<>?:"{}|]$/.test(key)

    const chord: string[] = []
    if (e.ctrlKey)  chord.push('Ctrl')
    if (e.shiftKey && !isSymbolFromShift) chord.push('Shift')
    if (e.altKey)   chord.push('Alt')
    if (!['Ctrl','Shift','Alt'].includes(key)) chord.push(key)
    return chord
  }
